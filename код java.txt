package com.example.excelapp;

import java.sql.*;
import java.util.ArrayList;
import java.util.List;

public class DatabaseManager {
    private Connection connection;

    public DatabaseManager() {
        connectToDatabase();
        createTableIfNotExists();
    }

    private void connectToDatabase() {
        try {
            connection = DriverManager.getConnection("jdbc:sqlite:performance_data.db");
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    private void createTableIfNotExists() {
        String sql = "CREATE TABLE IF NOT EXISTS performance_data (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
                number_n INTEGER NOT NULL,
                time_seconds REAL NOT NULL,
                result_number INTEGER NOT NULL
        );";
        try (Statement stmt = connection.createStatement()) {
            stmt.execute(sql);
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    public void insertData(DataModel data) {
        String sql = "INSERT INTO performance_data(number_n, time_seconds, result_number) VALUES(?, ?, ?)";
        try (PreparedStatement pstmt = connection.prepareStatement(sql)) {
            pstmt.setInt(1, data.getNumberN());
            pstmt.setDouble(2, data.getTimeSeconds());
            pstmt.setLong(3, data.getResultNumber());
            pstmt.executeUpdate();
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    public List<DataModel> getAllData() {
        List<DataModel> dataList = new ArrayList<>();
        String sql = "SELECT * FROM performance_data";
        try (Statement stmt = connection.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {

            while (rs.next()) {
                DataModel data = new DataModel();
                data.setId(rs.getInt("id"));
                data.setNumberN(rs.getInt("number_n"));
                data.setTimeSeconds(rs.getDouble("time_seconds"));
                data.setResultNumber(rs.getLong("result_number"));
                dataList.add(data);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return dataList;
    }

    public void deleteData(int id) {
        String sql = "DELETE FROM performance_data WHERE id = ?";
        try (PreparedStatement pstmt = connection.prepareStatement(sql)) {
            pstmt.setInt(1, id);
            pstmt.executeUpdate();
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

package com.example.excelapp;

public class DataModel {
    private int id;
    private int numberN;
    private double timeSeconds;
    private long resultNumber;

    public DataModel() {}

    public DataModel(int numberN, double timeSeconds, long resultNumber) {
        this.numberN = numberN;
        this.timeSeconds = timeSeconds;
        this.resultNumber = resultNumber;
    }

    // Геттеры и сеттеры
    public int getId() { return id; }
    public void setId(int id) { this.id = id; }

    public int getNumberN() { return numberN; }
    public void setNumberN(int numberN) { this.numberN = numberN; }

    public double getTimeSeconds() { return timeSeconds; }
    public void setTimeSeconds(double timeSeconds) { this.timeSeconds = timeSeconds; }

    public long getResultNumber() { return resultNumber; }
    public void setResultNumber(long resultNumber) { this.resultNumber = resultNumber; }
}
